const express = require('express');
const router = express.Router();

router.get('/', function (req, res) {
	res.render('javascript-and-jquery', { title: 'javascript-and-jquery' });

The jquery library (jquery-3.4.1.min.js) has been downloaded to the public/javascripts directory and is imported with the first script tag. A human readable version of the library can be seen at jquery.com.
